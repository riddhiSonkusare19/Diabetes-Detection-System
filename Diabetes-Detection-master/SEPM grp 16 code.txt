        // Implementation of a Random Forest-like model in JavaScript
        // Based on the features of the model in the GitHub repo
        
        // Feature normalization function
        function normalizeFeatures(features) {
            // Mean and std values approximated from the dataset
            const means = [3.8, 120.9, 69.1, 20.5, 79.8, 32.0, 0.47, 33.2];
            const stds = [3.4, 32.0, 19.4, 16.0, 115.2, 7.9, 0.33, 11.8];
            
            return features.map((val, idx) => (val - means[idx]) / stds[idx]);s
        }
        
        // Logistic function to convert to probability
        function sigmoid(x) {
            return 1 / (1 + Math.exp(-x));
        }
        
        // Random Forest model simulation
        function randomForestPredict(features) {
            // Normalize features
            const normalizedFeatures = normalizeFeatures(features);
            
            // Feature importance weights (based on typical Random Forest for diabetes prediction)
            // These weights approximate the importance of each feature in the model
            const weights = [0.07, 0.28, 0.06, 0.05, 0.04, 0.17, 0.12, 0.21];
            
            // Calculate weighted sum with some decision boundaries from random forest
            let score = 0;
            for (let i = 0; i < normalizedFeatures.length; i++) {
                score += normalizedFeatures[i] * weights[i];
            }
            
            // Add decision tree-like logic
            if (features[1] > 140) { // High glucose is most important factor
                score += 1.2;
            }
            
            if (features[5] > 30 && features[7] > 40) { // High BMI and age
                score += 0.8;
            }
            
            if (features[6] > 0.8) { // High diabetes pedigree
                score += 0.7;
            }
            
            // Convert to probability using sigmoid function
            return sigmoid(score - 0.6); // Threshold adjustment
        }
        
        document.getElementById('diabetesForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Show loader
            document.getElementById('loader').style.display = 'block';
            
            // Hide previous result
            document.getElementById('result').style.display = 'none';
            
            // Get all input values
            const pregnancies = parseFloat(document.getElementById('pregnancies').value);
            const glucose = parseFloat(document.getElementById('glucose').value);
            const bloodPressure = parseFloat(document.getElementById('bloodPressure').value);
            const skinThickness = parseFloat(document.getElementById('skinThickness').value);
            const insulin = parseFloat(document.getElementById('insulin').value);
            const bmi = parseFloat(document.getElementById('bmi').value);
            const diabetesPedigree = parseFloat(document.getElementById('diabetesPedigree').value);
            const age = parseFloat(document.getElementById('age').value);
            
            // Simple validation
            if (isNaN(pregnancies) || isNaN(glucose) || isNaN(bloodPressure) || 
                isNaN(skinThickness) || isNaN(insulin) || isNaN(bmi) || 
                isNaN(diabetesPedigree) || isNaN(age)) {
                
                // Hide loader
                document.getElementById('loader').style.display = 'none';
                
                showResult('Please fill in all fields with valid numbers.', 'error');
                return;
            }
            
            // Create feature array
            const features = [
                pregnancies, glucose, bloodPressure, skinThickness, 
                insulin, bmi, diabetesPedigree, age
            ];
            
            // Simulate API call delay
            setTimeout(() => {
                // Get prediction from model
                const probability = randomForestPredict(features);
                
                // Hide loader
                document.getElementById('loader').style.display = 'none';
                
                // Determine result based on probability
                if (probability >= 0.6) {
                    showResult(`Prediction: <strong>Higher Risk for Diabetes</strong>`, 'positive', probability);
                } else if (probability >= 0.4) {
                    showResult(`Prediction: <strong>Moderate Risk for Diabetes</strong>`, 'positive', probability);
                } else {
                    showResult(`Prediction: <strong>Lower Risk for Diabetes</strong>`, 'negative', probability);
                }
            }, 1500);
        });
        
        function showResult(message, type, probability = null) {
            const resultDiv = document.getElementById('result');
            const resultText = document.getElementById('resultText');
            resultText.innerHTML = message;
            resultDiv.className = 'result ' + type;
            resultDiv.style.display = 'block';
            
            // Show probability bar if probability is provided
            if (probability !== null) {
                const probabilityContainer = document.getElementById('probabilityContainer');
                const probabilityMarker = document.getElementById('probabilityMarker');
                const probabilityLabel = document.getElementById('probabilityLabel');
                
                probabilityContainer.style.display = 'block';
                probabilityMarker.style.left = `${probability * 100}%`;
                probabilityLabel.textContent = `Probability: ${(probability * 100).toFixed(1)}%`;
                
                // Additional explanation based on probability
                let explanation = '';
                if (probability >= 0.6) {
                    explanation = '<p>The model indicates a higher risk for diabetes. Key risk factors in your data may include elevated glucose levels, high BMI, age, and family history.</p>';
                } else if (probability >= 0.4) {
                    explanation = '<p>The model indicates a moderate risk for diabetes. Some risk factors are present, but they are not all severe.</p>';
                } else {
                    explanation = '<p>The model indicates a lower risk for diabetes based on the provided parameters.</p>';
                }
                
                explanation += '<p><strong>Remember:</strong> This is a simplified model for educational purposes. Always consult with healthcare professionals for proper medical advice.</p>';
                
                resultText.innerHTML += explanation;
            }
            
            // Scroll to result
            resultDiv.scrollIntoView({ behavior: 'smooth' });
        }
